<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>内置对象Date</title>

<body>
<script>
    /*
    window      窗口对象
    document    文档对象
    Array       数组对象
    String      字符串对象
    Boolean     布尔对象
    Date        日期对象
    Error       异常对象
    Math        数学对象
    Arguments   函数参数对象
    Object      基对象
    Regexp      正则对象
    Number      数值对象
    Function    函数构造器对象
     */
    /*
    Date对象的常用方法
        getDate()       获取日1-31
        getDay()        获取星期0-7
        getMonth()      获取月0-11
        getFullYear()   获取完整年份（浏览器都支持）
        getHours()      获取小时0-23
        getMinutes()    获取分钟0-59
        getSeconds()    获取秒0-59
        getMilliseconds()获取毫秒
        getTime()       返回累计毫秒数

     */
</script>
</body>
</html>
